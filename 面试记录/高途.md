输出所有和为N的连续正数序列.
输入一个正数N, 输出所有和为N的连续正数序列. 例如输入15, 结果: [[1, 2, 3, 4, 5], [4, 5, 6], [7, 8]]
```

 function fn(value) {
     if(value === 0) return []
     if(value === 1) return [[1]]
     const res = []
     const halfValue = Math.floor(value/2)+1
     for(let i = 1;i<halfValue;i++){
         let current = i+1
         let total = i+current
         let curArr = [i]
         let flag = false
          while(total<value+1&&!flag){
                if(total === value){
                    curArr.push(current)
                    res.push(curArr)
                    flag = true
                }else{
                    curArr.push(current)
                    current++
                    total = total+current
                }
             }
     }
     return res
 }
 fn(15)
```
