有一个任务队列，每项是一个函数，实现一个方法使得每项任务的间隔事件超过150ms,且该任务队列一直的在添加任务，js实现
```
class TaskQueue {
    constructor(minInterval = 150) {
        this.queue = [];
        this.minInterval = minInterval;
        this.isProcessing = false;
        this.lastExecutionTime = 0;
    }

    // 添加任务到队列
    addTask(task) {
        if (typeof task !== 'function') {
            throw new Error('Task must be a function');
        }
        
        this.queue.push(task);
        this.processQueue();
    }

    // 处理队列
    async processQueue() {
        if (this.isProcessing || this.queue.length === 0) {
            return;
        }

        this.isProcessing = true;

        while (this.queue.length > 0) {
            const currentTime = Date.now();
            const timeSinceLastExecution = currentTime - this.lastExecutionTime;
            
            // 计算需要等待的时间
            const waitTime = Math.max(0, this.minInterval - timeSinceLastExecution);
            
            if (waitTime > 0) {
                await this.delay(waitTime);
            }

            // 执行任务
            const task = this.queue.shift();
            try {
                task();
            } catch (error) {
                console.error('Task execution error:', error);
            }
            
            this.lastExecutionTime = Date.now();
        }

        this.isProcessing = false;
    }

    // 延迟函数
    delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }

    // 获取队列长度
    getQueueLength() {
        return this.queue.length;
    }

    // 清空队列
    clearQueue() {
        this.queue = [];
    }
}
```
